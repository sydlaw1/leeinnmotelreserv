-- Create Tables
-- =============================================
-- Lee Inn and Suites — MS SQL Server Script
-- Tables: Guests, Rooms, Reservations
-- =============================================

CREATE TABLE Guests (
    GuestID    INT           IDENTITY(1,1)  PRIMARY KEY,
    LastName   VARCHAR(50)   NOT NULL,
    FirstName  VARCHAR(50)   NOT NULL,
    Address    VARCHAR(100),
    City       VARCHAR(50),
    State      CHAR(2)
);

CREATE TABLE Rooms (
    RoomNumber INT           PRIMARY KEY,
    Floor      TINYINT       NOT NULL,
    RoomType   CHAR(2)       NOT NULL,
    CONSTRAINT chk_type CHECK (RoomType IN ('QQ', 'K', 'Q'))
);

CREATE TABLE Reservations (
    ReservationID INT    IDENTITY(1001,1) PRIMARY KEY,
    GuestID       INT    NOT NULL
                          REFERENCES Guests(GuestID),
    RoomNumber    INT    NOT NULL
                          REFERENCES Rooms(RoomNumber),
    CheckInDate   DATE   NOT NULL,
    CheckOutDate  DATE   NOT NULL,
    CONSTRAINT chk_dates
        CHECK (CheckOutDate > CheckInDate)
);

-- Search / Sort Query
-- Search reservations by guest last name, sorted by check-in
SELECT
    r.ReservationID,
    g.LastName  AS LastName,
    g.FirstName AS FirstName,
    rm.RoomNumber,
    rm.RoomType,
    r.CheckInDate,
    r.CheckOutDate,
    DATEDIFF(day, r.CheckInDate, r.CheckOutDate) AS Nights
FROM        Reservations r
INNER JOIN  Guests       g  ON r.GuestID    = g.GuestID
INNER JOIN  Rooms        rm ON r.RoomNumber = rm.RoomNumber
WHERE       g.LastName LIKE '%Lee%'
ORDER BY    r.CheckInDate ASC;
